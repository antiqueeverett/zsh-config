}

#
# todo --- remove this !!!! opening *.pdf and *.md files
#
function opn() {
    FILE="$1"
    echo "-- opening $FILE"

    if [ "${FILE##*.}" = "md" ]; then
        pandoc $FILE | lynx -stdin

    elif [ "${FILE##*.}" = "pdf" ]; then
        if [[ "$OSTYPE" == "linux-gnu" ]]; then
            xdg-open "$FILE"

        elif [[ "$OSTYPE" == "darwin"* ]]; then
            open "$FILE"
        fi
    fi
}
